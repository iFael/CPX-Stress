**1. Add `buildVerdict()` helper function** — place it near `buildFindings()` (around line 872)

```typescript
function buildVerdict(
  result: TestResult,
  health: HealthAssessment,
): { sentence: string; contextNote: string | null } {
  const vus = result.config.virtualUsers;
  const avgMs = formatMs(result.latency.avg);
  const errPct = result.errorRate.toFixed(1);

  let sentence: string;

  if (result.errorRate < 5) {
    sentence = `O sistema suportou ${vus} usuários simultâneos com tempo de resposta médio de ${avgMs} e taxa de erro de ${errPct}%.`;
  } else if (result.errorRate < 20) {
    sentence = `O sistema apresentou dificuldades com ${vus} usuários simultâneos: tempo de resposta médio de ${avgMs} e taxa de erro de ${errPct}%.`;
  } else {
    sentence = `O sistema não suportou ${vus} usuários simultâneos adequadamente: tempo de resposta médio de ${avgMs} e ${errPct}% das requisições falharam.`;
  }

  const contextNote =
    result.errorRate > 5
      ? "Nota: É comum que servidores web apresentem aumento de erros quando o número de acessos simultâneos ultrapassa a capacidade de processamento configurada. Isso pode ser ajustado pela equipe de infraestrutura."
      : null;

  return { sentence, contextNote };
}
```
